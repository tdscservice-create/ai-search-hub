<script>
    async function searchAICloud() {
        const query = document.getElementById('searchInput').value;
        const categoryFilter = document.getElementById('categorySelect').value;
        const searchStatus = document.getElementById('searchStatus');
        const resultsDiv = document.getElementById('results');
        if (!query.trim()) return alert("Please enter a search phrase.");
        searchStatus.innerText = "Analyzing unknown folder documents...";
        resultsDiv.innerHTML = "";
        try {
            const response = await fetch('/api/search-ai-data', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ query, categoryFilter })
            });
            const data = await response.json();
            searchStatus.innerText = "";
            if (data.results && data.results.length > 0) {
                data.results.forEach(item => {
                    const div = document.createElement('div');
                    div.className = 'result-item';
                    div.innerHTML = `
                        <span class="tag">${item.category.toUpperCase()}</span>
                        <p class="file-title">📄 Found in File: ${item.file_name || 'Unknown Document'}</p>
                        <p class="summary"><strong>AI Summary of Content:</strong> ${item.summary}</p>
                        <div style="font-size:14px; line-height:1.4; color:#444; background:#f0f0f0; padding:10px; border-radius:4px;">...${item.content.substring(0, 250)}...</div>
                        <span style="font-size:11px; color:#999; display:block; margin-top:10px;">Match Confidence: ${Math.round(item.similarity * 100)}%</span>
                    `;
                    resultsDiv.appendChild(div);
                });
            } else {
                resultsDiv.innerHTML = "<p style='text-align:center; font-size:14px;'>No matching information found inside the folder.</p>";
            }
        } catch (err) { searchStatus.innerText = "❌ Search failed."; }
    }
</script>
